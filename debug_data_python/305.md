def getFibonacci(n):
       if n<3:
        return 1 
       else:return getFibonacci(n-1)+getFibonacci (n-2)
N =int(input())
for i in range(N):
    line = list(map(int,input().split(", ")))
    n=line[0]
    d=line[1]
    s=[]
    for j in range(1,n):
