```python
def bisection(f_str, a, b, eps):
    """二分法求方程的根"""
    def f(x):
        return eval(f_str)
    if a > b:
        a, b = b, a
    if f(a) * f(b) > 0:
        print('方程在[a, b]区间内无解')
    else:
        n = 0
        while True:
            n += 1
            root = (a+b)/2
            y = f(root)
            if abs(y) < eps:
                break
            if f(a) * y < 0:
                b = root
            else:
                a = root
        print('经过%d次迭代求得方程的根为%f'%(n, root))
```
