import random
xitong=random.randint(0,2)
if xitong==0:
    xitong="剪刀"
else:
    if xitong==1:
        xitong="石头"
    else:
        xitong="布"
while True:
    people=input("请你输入石头/剪刀/布:")
    if people != '剪刀' or people != '石头' or people != '布':
        print('你的输入有误, 请重新开始!')
        people = input ( "请你输入石头/剪刀/布:" )
    if xitong==people:
        print("系统:",xitong)
        print("平局")
    elif (xitong=="石头" and people=="布") or (xitong=="布" and people=="剪刀") or (xitong=="剪刀" and people=="石头"):
        print("系统:",xitong)
        print("你赢了")

    else:
